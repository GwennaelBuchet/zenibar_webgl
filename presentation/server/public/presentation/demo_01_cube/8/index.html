2. Indiquer Ã  WebGL comment lire les Buffers

    gl.bindBuffer(gl.ARRAY_BUFFER, positionsBuffer);
    gl.vertexAttribPointer(
        0,          // index of input location
        3,          // size
        gl.FLOAT,   // type
        false,      // normalized
        0,          // stride (jump to do between reads)
        0           // offset (starting point)
    );
    gl.enableVertexAttribArray(0);
